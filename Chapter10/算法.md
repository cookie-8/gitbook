# 第 1 节：算法

1. 斐波那契变种
   fn(2) => [1,1,2,3,5,8,13]
   fn(3) => [1,1,1,3,5,9,14]
   fn(4) => [1,1,1,1,4,7,13,25]
   求 fn(n)中的第 index 个元素
   fn(key,index)

```js

function fn(key, index) {
  const map = {}
  function fib(n) {
    if (map[n]) return map[n]
    if (n < key) {
      map[n] = 1
      return 1
    }

    let sum = 0
    for (let i = 1; i <= key; i++) {
      sum += fib(n - i)
    }
    map[n] = sum
    return sum
  }
  return fib(index)
}
```
